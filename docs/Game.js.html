/**
 * 比赛模块
 * @class
 */
class Game {
  
  /**
   * 构建函数
   * @param  {string} x [description]
   * @param  {string} y [description]
   * @return {void}   [description]
   */
  constructor(x, y) {
    this.x = x;
    this.y = y;
  }
  
  /**
   * * 转换字符串
   * @return {string} 组合语句
   */
  toString() {
    return '(' + this.x + ', ' + this.y + ')';
  }

  /**
   * 开始比赛
   * @return {string} 提示比赛开始了
   */
  play() {
    return "开始比赛了"
  }

}
